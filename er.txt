    retention_categories {
        UUID id PK "default gen_random_uuid()"
        VARCHAR_64 code UK "NOT NULL, UNIQUE"
        VARCHAR_255 name "NOT NULL"
        TEXT description
        VARCHAR_16 status "NOT NULL, DEFAULT 'draft', CHECK(active, draft, inactive)"
        BOOLEAN has_ever_held_content "NOT NULL, DEFAULT FALSE"
        TIMESTAMPTZ created_at "NOT NULL, DEFAULT NOW()"
        VARCHAR_255 created_by "NOT NULL"
        TIMESTAMPTZ updated_at "NOT NULL, DEFAULT NOW()"
        VARCHAR_255 updated_by "NOT NULL"
        TIMESTAMPTZ deleted_at
        VARCHAR_255 deleted_by
    }

    retention_sub_categories {
        UUID id PK "default gen_random_uuid()"
        UUID category_id FK "NOT NULL, REFERENCES txn.retention_categories(id)"
        VARCHAR_64 code "NOT NULL"
        VARCHAR_255 name "NOT NULL"
        TEXT description
        INTEGER retention_duration_value "NOT NULL, CHECK(> 0)"
        VARCHAR_16 retention_duration_unit "NOT NULL, CHECK(days, months, years)"
        VARCHAR_16 status "NOT NULL, DEFAULT 'draft', CHECK(active, draft, inactive)"
        BOOLEAN has_ever_held_content "NOT NULL, DEFAULT FALSE"
        TIMESTAMPTZ created_at "NOT NULL, DEFAULT NOW()"
        VARCHAR_255 created_by "NOT NULL"
        TIMESTAMPTZ updated_at "NOT NULL, DEFAULT NOW()"
        VARCHAR_255 updated_by "NOT NULL"
        TIMESTAMPTZ deleted_at
        VARCHAR_255 deleted_by
    }

    retention_documents {
        UUID id PK "default gen_random_uuid()"
        UUID sub_category_id FK "NOT NULL, REFERENCES txn.retention_sub_categories(id)"
        VARCHAR_64 source_system "NOT NULL"
        VARCHAR_255 source_reference "NOT NULL"
        VARCHAR_512 filename "NOT NULL"
        BIGINT size_bytes "NOT NULL, CHECK(> 0)"
        CHAR_64 sha256 "NOT NULL"
        VARCHAR_255 uploaded_by_user_id "NOT NULL"
        DATE basis_date "NOT NULL"
        DATE eligibility_date
        TIMESTAMPTZ retention_evaluated_at
        TIMESTAMPTZ created_at "NOT NULL, DEFAULT NOW()"
        VARCHAR_255 created_by "NOT NULL"
        TIMESTAMPTZ updated_at "NOT NULL, DEFAULT NOW()"
        VARCHAR_255 updated_by "NOT NULL"
        TIMESTAMPTZ deleted_at
        VARCHAR_255 deleted_by
    }

    retention_table_onboarding {

        TEXT_ARRAY primary_key_columns "NOT NULL"
        VARCHAR_255 basis_date_column "NOT NULL"
        UUID default_sub_category_id FK "NOT NULL, REFERENCES txn.retention_sub_categories(id)"
        VARCHAR_255 registered_by "NOT NULL"
        TIMESTAMPTZ registered_at "NOT NULL, DEFAULT NOW()"
        TIMESTAMPTZ created_at "NOT NULL, DEFAULT NOW()"
        VARCHAR_255 created_by "NOT NULL"
        TIMESTAMPTZ updated_at "NOT NULL, DEFAULT NOW()"
        VARCHAR_255 updated_by "NOT NULL"
        TIMESTAMPTZ deleted_at
        VARCHAR_255 deleted_by
    }
